                    let count = 0;

                    const updateCount = () => {
                        if (count < countTo) {
                            count += increment;
                            target.innerText = Math.ceil(count);
                            setTimeout(updateCount, 20);
                        } else {
                            target.innerText = countTo;
                        }
                    };
                    updateCount();
                    countObserver.unobserve(target);
                }
            });
        }, { threshold: 0.2 });

        counters.forEach(c => countObserver.observe(c));

        // 7. Dynamic Mouse Spotlight Tracker (Liquid UI Style - Global Overlay)
        const globalSpotlight = document.getElementById('global-spotlight');
        document.addEventListener('mousemove', e => {
            if (globalSpotlight) {
                globalSpotlight.style.opacity = '1';
                if (document.body.classList.contains('light-theme')) {
                    globalSpotlight.style.mixBlendMode = 'multiply';
                    globalSpotlight.style.background = `radial-gradient(800px circle at ${e.clientX}px ${e.clientY}px, rgba(0, 48, 86, 0.08), transparent 40%)`;
                } else {
                    globalSpotlight.style.mixBlendMode = 'overlay';
                    globalSpotlight.style.background = `radial-gradient(800px circle at ${e.clientX}px ${e.clientY}px, rgba(255, 255, 255, 0.15), transparent 40%)`;
                }
            }
        });
        document.addEventListener('mouseleave', () => {
            if (globalSpotlight) globalSpotlight.style.opacity = '0';
        });
    </script>
